def getEmisionGEI
  aux = listaAlimentos.head
  aux2 = listaGramos.head
  
  geiTotal = 0
  while aux != nil && aux2 != nil
    geiTotal += aux.value.GEI * (aux2.value / 100)
    aux = aux.next
    aux2 = aux2.next
  end
  geiTotal.round(2)
end
